{{- else if contains "LoadBalancer" .Values.service.type }}

1. Get the application URL by running:

   NOTE: It may take a few minutes for the LoadBalancer IP to be available.

   kubectl get svc --namespace {{ .Release.Namespace }} {{ include "llm-config-manager.fullname" . }} -w

   export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "llm-config-manager.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')

   echo "Application URL: http://$SERVICE_IP:{{ .Values.service.port }}"

{{- else if contains "ClusterIP" .Values.service.type }}

1. Access the application by port-forwarding:

   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "llm-config-manager.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}

   echo "Application URL: http://127.0.0.1:{{ .Values.service.port }}"

{{- end }}

2. Check the application health:

   curl http://{{ if .Values.ingress.enabled }}{{ (index .Values.ingress.hosts 0).host }}{{ else }}127.0.0.1:{{ .Values.service.port }}{{ end }}/health

3. View application logs:

   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "llm-config-manager.name" . }} --tail=100 -f

4. Access metrics endpoint:

{{- if .Values.ingress.enabled }}
   http{{ if $.Values.ingress.tls }}s{{ end }}://{{ (index .Values.ingress.hosts 0).host }}:{{ .Values.service.metricsPort }}/metrics
{{- else }}
   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "llm-config-manager.fullname" . }} {{ .Values.service.metricsPort }}:{{ .Values.service.metricsPort }}
   curl http://127.0.0.1:{{ .Values.service.metricsPort }}/metrics
{{- end }}
